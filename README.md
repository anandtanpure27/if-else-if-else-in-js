# if-else-if-else-in-js
this is a simple java scrip program to calculate popcorn price using if-else.
let size = "m";
if(size === "xl"){
    console.log("the price of popcorn is 250 rupees");
}
else if(size === "l"){
    console.log("the price of popcorn is 200 rupees");
}
else if(size === "m"){
    console.log("the price  of popcorn is 150 rupees");
}
else{
    console.log("the price of popcorn is 50 rupees");
}
